func quickSort(arr []int) []int {
	if len(arr) < 2 {
		return arr
	}
	left := []int{}
	right := []int{}
	pivot := arr[0]
	for _, value := range arr[1:] {
		if value <= pivot {
			left = append(left, value)
		} else {
			right = append(right, value)
		}
	}
	return append(append(quickSort(left), pivot), quickSort(right)...)
}
